presentation error问题

题目要求格式必须正确，如果在输出结果中，多了或少了不必要的空格或者回车或者其他，就会有这种错误。

常见解决办法如下：

最后输入必须换行：

int count = 0;
if(满足题目要求) count++;
最终输出：
    if(count > 0) printf(" ");
	printf("%d", ans);//此处无需加空格
	else if(count == 0) printf("xxx");//根据题目要求输出，比如no
一般最后还需要换行。

比如下面这道水仙花的题目的这样一个presentation error 就让我debug好久，因此，这里记录一下，以免下次又遇到类似的问题无从下手。

#include<iostream>
#include<vector>
#include<algorithm>
#include<math.h>
using namespace std;
float a[1005];
#define PI 3.1415927


int main() {
	int m, n;
	
	while (~scanf("%d%d", &m, &n)) {
		bool ok = false;
		int count = 0;
		for (int i = m; i <= n; i++) {
			int hun = i / 100;
			int ten = (i - hun * 100) / 10;
			int ind = i % 10;
			if ((hun*hun*hun)+(ten*ten*ten)+(ind*ind*ind) == i) {
				if (count > 0) printf(" ");
				printf("%d", i);
				ok = true;
				count++;
			}
		}if (count == 0) printf("no");
		printf("\n");
	}
	return 0;
}

第三个技巧：多组输入，对应的值不能为0

比如下面这道水题， 开始一直presentation error 后来我在多组输入while循环里面加了一个&& n （表示n）不能为0，立马就A了。

错误示范

while (~scanf("%d", &n) )

正确示范

while (~scanf("%d", &n) && n)

#include<iostream>
#include<vector>
#include<algorithm>
#include<math.h>
#include<stdio.h>
using namespace std;
int a[1005], b[108];
int sum, cnt;
#define PI 3.1415927

bool cmp(int a, int b) {
	return a > b;
}
int main() {
	int n;
	while (~scanf("%d", &n) && n) {
		for (int i = 0; i < n; i++) {
			scanf("%d", &a[i]);
			b[i] = abs(a[i]);
		}
		sort(b, b + n, cmp);
		int count = 0;
		for (int i = 0; i < n; i++) {
			for(int j = 0; j < n; j++)
				if (b[i] == abs(a[j])) {
					if (i != 0) printf(" ");
					printf("%d", a[j]);
			}
		}
		printf("\n");
	}
	return 0;
}
